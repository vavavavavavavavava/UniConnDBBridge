    # utils/crud.py
from typing import Type, List, Dict, Any, Optional
from sqlalchemy import select, insert, update, delete
from sqlalchemy.orm import DeclarativeBase

class CRUDMixin:
    """CRUD 操作ミックスイン"""

    def __init__(self, manager: Union[DatabaseManager, AsyncDatabaseManager]):
        self.manager = manager

    # 同期版 CRUD
    def create(
        self,
        model: Type[DeclarativeBase],
        **values: Any
    ) -> DeclarativeBase:
        """レコード作成"""
        with self.manager.session() as session:
            instance = model(**values)
            session.add(instance)
            session.flush()
            session.refresh(instance)
            return instance

    def read(
        self,
        model: Type[DeclarativeBase],
        **filters: Any
    ) -> Optional[DeclarativeBase]:
        """単一レコード読み取り"""
        with self.manager.session() as session:
            stmt = select(model).filter_by(**filters)
            return session.execute(stmt).scalar_one_or_none()

    def read_many(
        self,
        model: Type[DeclarativeBase],
        limit: Optional[int] = None,
        offset: Optional[int] = None,
        **filters: Any
    ) -> List[DeclarativeBase]:
        """複数レコード読み取り"""
        with self.manager.session() as session:
            stmt = select(model).filter_by(**filters)
            if limit:
                stmt = stmt.limit(limit)
            if offset:
                stmt = stmt.offset(offset)
            return list(session.execute(stmt).scalars())

    def update(
        self,
        model: Type[DeclarativeBase],
        values: Dict[str, Any],
        **filters: Any
    ) -> int:
        """レコード更新"""
        with self.manager.session() as session:
            stmt = update(model).filter_by(**filters).values(**values)
            result = session.execute(stmt)
            return result.rowcount

    def delete(
        self,
        model: Type[DeclarativeBase],
        **filters: Any
    ) -> int:
        """レコード削除"""
        with self.manager.session() as session:
            stmt = delete(model).filter_by(**filters)
            result = session.execute(stmt)
            return result.rowcount


class AsyncCRUDMixin:
    """非同期 CRUD 操作ミックスイン"""

    def __init__(self, manager: AsyncDatabaseManager):
        self.manager = manager

    async def create(
        self,
        model: Type[DeclarativeBase],
        **values: Any
    ) -> DeclarativeBase:
        """非同期レコード作成"""
        async with self.manager.session() as session:
            instance = model(**values)
            session.add(instance)
            await session.flush()

    6.2 OpenTelemetry 統合

    # telemetry.py
from typing import Optional, Dict, Any
from contextlib import contextmanager
from opentelemetry import trace
from opentelemetry.trace import Tracer, Span
from opentelemetry.instrumentation.sqlalchemy import SQLAlchemyInstrumentor

class TelemetryManager:
    """テレメトリ管理"""

    def __init__(self, service_name: str = "uniconndbbridge"):
        self.service_name = service_name
        self.tracer: Optional[Tracer] = None
        self._instrumentor: Optional[SQLAlchemyInstrumentor] = None

    def initialize(
        self,
        endpoint: Optional[str] = None,
        headers: Optional[Dict[str, str]] = None
    ) -> None:
        """テレメトリ初期化"""
        from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
        from opentelemetry.sdk.trace import TracerProvider
        from opentelemetry.sdk.trace.export import BatchSpanProcessor
        from opentelemetry.sdk.resources import Resource

        # リソース設定
        resource = Resource.create({
            "service.name": self.service_name,
            "service.version": "25.6.0"
        })

        # プロバイダー設定
        provider = TracerProvider(resource=resource)

        # エクスポーター設定
        if endpoint:
            exporter = OTLPSpanExporter(
                endpoint=endpoint,
                headers=headers
            )
            processor = BatchSpanProcessor(exporter)
            provider.add_span_processor(processor)

        # グローバル設定
        trace.set_tracer_provider(provider)
        self.tracer = trace.get_tracer(__name__)

        # SQLAlchemy インストルメンテーション
        self._instrumentor = SQLAlchemyInstrumentor()

    def instrument_engine(self, engine: Engine) -> None:
        """Engine にテレメトリを適用"""
        if self._instrumentor:
            self._instrumentor.instrument(
                engine=engine,
                service=self.service_name
            )

    @contextmanager
    def span(self, name: str, **attributes: Any):
        """スパンコンテキストマネージャー"""
        if not self.tracer:
            yield None
            return

        with self.tracer.start_as_current_span(name) as span:
            for key, value in attributes.items():
                span.set_attribute(key, str(value))
            yield span

    8.1 接続プーリング最適化

    # optimizations/pooling.py
from typing import Dict, Any
from sqlalchemy.pool import QueuePool, NullPool, StaticPool

class PoolOptimizer:
    """プール最適化ヘルパー"""

    @staticmethod
    def get_optimal_pool_class(dialect: str, is_async: bool = False):
        """方言に最適なプールクラスを選択"""

        if dialect == "sqlite":
            # SQLite はシングルスレッドなので StaticPool
            return StaticPool
        elif is_async:
            # 非同期は通常 NullPool が推奨
            return NullPool
        else:
            # その他は QueuePool
            return QueuePool

    @staticmethod
    def get_optimal_pool_size(dialect: str, expected_load: str = "medium") -> Dict[str, int]:
        """負荷に応じた最適なプールサイズ"""

        configs = {
            "low": {"pool_size": 5, "max_overflow": 5},
            "medium": {"pool_size": 10, "max_overflow": 10},
            "high": {"pool_size": 20, "max_overflow": 20},
            "extreme": {"pool_size": 50, "max_overflow": 50}
        }

        base_config = configs.get(expected_load, configs["medium"])

        # 方言固有の調整
        if dialect == "oracle":
            # Oracle は接続コストが高いのでプールサイズを増やす
            base_config["pool_size"] = int(base_config["pool_size"] * 1.5)
        elif dialect == "sqlite":
            # SQLite は接続が軽量
            base_config["pool_size"] = 1
            base_config["max_overflow"] = 0

        return base_config

from typing import List, Any
from sqlalchemy import text
from sqlalchemy.sql import Select

class QueryOptimizer:
    """クエリ最適化ヘルパー"""

    @staticmethod
    def add_execution_options(stmt: Select, dialect: str) -> Select:
        """方言固有の実行オプション追加"""

        if dialect == "postgresql":
            # PostgreSQL 固有の最適化
            return stmt.execution_options(
                postgresql_readonly=True,
                postgresql_deferrable=True
            )
        elif dialect == "mysql":
            # MySQL 固有の最適化
            return stmt.execution_options(
                mysql_buffered=True
            )

        return stmt

    @staticmethod
    def optimize_bulk_insert(
        values: List[Dict[str, Any]],
        dialect: str
    ) -> Dict[str, Any]:
        """バルクインサート最適化"""

        if dialect == "postgresql":
            # PostgreSQL は COPY が最速
            return {
                "method": "copy",
                "chunk_size": 10000
            }
        elif dialect == "mysql":
            # MySQL は multi-value insert
            return {
                "method": "multi",
                "chunk_size": 5000
            }
        else:
            # デフォルト
            return {
                "method": "multi",
                "chunk_size": 1000
            }

    9.1 認証情報の保護

    # security/credentials.py
import os
from typing import Optional
from cryptography.fernet import Fernet

class CredentialManager:
    """認証情報管理"""

    def __init__(self, key: Optional[bytes] = None):
        if key is None:
            key = os.getenv("UNICONNDB_ENCRYPTION_KEY")
            if not key:
                key = Fernet.generate_key()

        self.cipher = Fernet(key)

    def encrypt_password(self, password: str) -> str:
        """パスワード暗号化"""
        return self.cipher.encrypt(password.encode()).decode()

    def decrypt_password(self, encrypted: str) -> str:
        """パスワード復号"""
        return self.cipher.decrypt(encrypted.encode()).decode()

    @staticmethod
    def mask_url(url: str) -> str:
        """URL のパスワード部分をマスク"""
        from urllib.parse import urlparse, urlunparse

        parsed = urlparse(url)
        if parsed.password:
            # パスワードを *** に置換
            masked_netloc = parsed.netloc.replace(
                f":{parsed.password}@",
                ":***@"
            )
            masked = parsed._replace(netloc=masked_netloc)
            return urlunparse(masked)

        return url

    9.2 SQL インジェクション対策

    # security/validation.py
import re
from typing import Any

class SQLValidator:
    """SQL バリデーション"""

    # 危険なパターン
    DANGEROUS_PATTERNS = [
        r";\s*(DROP|DELETE|TRUNCATE|UPDATE|INSERT)",
        r"--",
        r"/\*.*\*/",
        r"xp_cmdshell",
        r"exec\s*\(",
        r"UNION\s+SELECT"
    ]

    @classmethod
    def is_safe_identifier(cls, identifier: str) -> bool:
        """識別子の安全性チェック"""
        # 英数字とアンダースコアのみ許可
        return bool(re.match(r"^[a-zA-Z_][a-zA-Z0-9_]*$", identifier))

    @classmethod
    def validate_query(cls, query: str) -> bool:
        """クエリの危険性チェック"""
        for pattern in cls.DANGEROUS_PATTERNS:
            if re.search(pattern, query, re.IGNORECASE):
                return False
        return True

    @classmethod
    def escape_identifier(cls, identifier: str, dialect: str) -> str:
        """識別子のエスケープ"""
        if dialect == "postgresql":
            return f'"{identifier}"'
        elif dialect == "mysql":
            return f"`{identifier}`"
        elif dialect == "mssql":
            return f"[{identifier}]"
        else:
            return identifier
