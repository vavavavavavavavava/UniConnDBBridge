class CRUDMixin:
    """CRUD 操作ミックスイン"""
    
    def __init__(self, manager: Union[DatabaseManager, AsyncDatabaseManager]):
        self.manager = manager
    
    # 同期版 CRUD
    def create(
        self,
        model: Type[DeclarativeBase],
        **values: Any
    ) -> DeclarativeBase:
        """レコード作成"""
        with self.manager.session() as session:
            instance = model(**values)
            session.add(instance)
            session.flush()
            session.refresh(instance)
            return instance
    
    def read(
        self,
        model: Type[DeclarativeBase],
        **filters: Any
    ) -> Optional[DeclarativeBase]:
        """単一レコード読み取り"""
        with self.manager.session() as session:
            stmt = select(model).filter_by(**filters)
            return session.execute(stmt).scalar_one_or_none()
    
    def read_many(
        self,
        model: Type[DeclarativeBase],
        limit: Optional[int] = None,
        offset: Optional[int] = None,
        **filters: Any
    ) -> List[DeclarativeBase]:
        """複数レコード読み取り"""
        with self.manager.session() as session:
            stmt = select(model).filter_by(**filters)
            if limit:
                stmt = stmt.limit(limit)
            if offset:
                stmt = stmt.offset(offset)
            return list(session.execute(stmt).scalars())
    
    def update(
        self,
        model: Type[DeclarativeBase],
        values: Dict[str, Any],
        **filters: Any
    ) -> int:
        """レコード更新"""
        with self.manager.session() as session:
            stmt = update(model).filter_by(**filters).values(**values)
            result = session.execute(stmt)
            return result.rowcount
    
    def delete(
        self,
        model: Type[DeclarativeBase],
        **filters: Any
    ) -> int:
        """レコード削除"""
        with self.manager.session() as session:
            stmt = delete(model).filter_by(**filters)

from contextlib import contextmanager
from opentelemetry import trace

from opentelemetry.instrumentation.sqlalchemy import SQLAlchemyInstrumentor

class TelemetryManager:

        headers: Optional[Dict[str, str]] = None
    ) -> None:

        from opentelemetry.sdk.trace import TracerProvider
        from opentelemetry.sdk.trace.export import BatchSpanProcessor
        from opentelemetry.sdk.resources import Resource
        
        # リソース設定
        resource = Resource.create({
            "service.name": self.service_name,
            "service.version": "25.6.0"
        })
        
        # プロバイダー設定
        provider = TracerProvider(resource=resource)
        
        # エクスポーター設定
        if endpoint:
            exporter = OTLPSpanExporter(
                endpoint=endpoint,
                headers=headers
            )
            processor = BatchSpanProcessor(exporter)
            provider.add_span_processor(processor)
        
        # グローバル設定
        trace.set_tracer_provider(provider)
        self.tracer = trace.get_tracer(__name__)
        
        # SQLAlchemy インストルメンテーション
        self._instrumentor = SQLAlchemyInstrumentor()
    
    def instrument_engine(self, engine: Engine) -> None:
        """Engine にテレメトリを適用"""
        if self._instrumentor:
            self._instrumentor.instrument(
                engine=engine,
                service=self.service_name
            )
    
    @contextmanager
    def span(self, name: str, **attributes: Any):
        """スパンコンテキストマネージャー"""
        if not self.tracer:
            yield None
            return
            
        with self.tracer.start_as_current_span(name) as span:
            for key, value in attributes.items():
                span.set_attribute(key, str(value))
            yield span

class PoolOptimizer:
    """プール最適化ヘルパー"""
    
    @staticmethod
    def get_optimal_pool_class(dialect: str, is_async: bool = False):
        """方言に最適なプールクラスを選択"""

        elif is_async:

            return NullPool

            "low": {"pool_size": 5, "max_overflow": 5},

            "high": {"pool_size": 20, "max_overflow": 20},

# security/validation.py
import re
from typing import Any

class SQLValidator:
    """SQL バリデーション"""
    
    # 危険なパターン
    DANGEROUS_PATTERNS = [
        r";\s*(DROP|DELETE|TRUNCATE|UPDATE|INSERT)",
        r"--",
        r"/\*.*\*/",
        r"xp_cmdshell",
        r"exec\s*\(",
        r"UNION\s+SELECT"
    ]
    
    @classmethod
    def is_safe_identifier(cls, identifier: str) -> bool:
        """識別子の安全性チェック"""
        # 英数字とアンダースコアのみ許可
        return bool(re.match(r"^[a-zA-Z_][a-zA-Z0-9_]*$", identifier))
    
    @classmethod
    def validate_query(cls, query: str) -> bool:
        """クエリの危険性チェック"""
        for pattern in cls.DANGEROUS_PATTERNS:
            if re.search(pattern, query, re.IGNORECASE):
                return False
        return True
    
    @classmethod
    def escape_identifier(cls, identifier: str, dialect: str) -> str:
        """識別子のエスケープ"""
        if dialect == "postgresql":
            return f'"{identifier}"'
        elif dialect == "mysql":
            return f"`{identifier}`"
        elif dialect == "mssql":
            return f"[{identifier}]"
        else:
            return identifier
